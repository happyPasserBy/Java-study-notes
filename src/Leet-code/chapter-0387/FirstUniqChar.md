案例:
s = "leetcode"

s = "loveleetcode",

public class Chapter {

    public int firstUniqChar(String s) {
        Map<Character,Integer> map = new HashMap<>();
        for (int i = 0; i < s.length(); i++) {
            Character key = s.charAt(i);
            map.put(key, map.getOrDefault(key,0)+1);
        }

        for (int i = 0; i < s.length(); i++) {
            Character key = s.charAt(i);
            if(map.get(key) == 1) {
                return i;
            }
        }

        return -1;
    }
